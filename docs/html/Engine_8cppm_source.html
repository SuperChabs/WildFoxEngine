116 if (viewportSize.x > 0 && viewportSize.y > 0)

118 GetRenderer()->Render(

120 static_cast<int>(viewportSize.x),
121 static_cast<int>(viewportSize.y)
122 );
123 }
124
125 GetRenderer()->EndFrame();
126
127 fb->Unbind();
128 }
129
130 glBindFramebuffer(GL_FRAMEBUFFER, 0);
